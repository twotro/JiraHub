local Material = loadstring(game:HttpGet("https://raw.githubusercontent.com/Kinlei/MaterialLua/master/Module.lua"))()

local UI = Material.Load({
     Title = "JiraHub - Project Upright (Very Buggy Rn)",
     Style = 2,
     SizeX = 400,
     SizeY = 400,
     Theme = "Dark"
})

local Page = UI.New({
    Title = "Main"
})

Page.Toggle({
    Text = "Auto-Punch",
    Callback = function(autoPunch)
        if autoPunch then
            _G.punch = true do
                spawn(function ()
                    while _G.punch == true do
                        local args = {
                            [1] = "Punch",
                            [2] = {}
                        }                   
                        game:GetService("ReplicatedStorage").Ability:FireServer(unpack(args))
                        wait(0.1)
                    end                  
                end)
            end
        else
            _G.punch = false do
                spawn(function ()
                    while _G.punch == false do
                        wait(1)
                  end
                end)
            end
        end
    end,
    Enabled = false
})
--Tween Pick Arrow
Page.Toggle({
    Text = "AutoPick-Arrow",
    Callback = function(autoPickArrow)
        if autoPickArrow then
            _G.autoPickArrow = true do
                spawn(function ()
                    while _G.autoPickArrow == true do
                        local New_CFrame = game:GetService("Workspace")["Stand Arrow"].CFrame
local ts = game:GetService("TweenService")
local char = game.Players.LocalPlayer.Character
local part = char.HumanoidRootPart
local ti = TweenInfo.new(3, Enum.EasingStyle.Linear)
local tp = {CFrame = New_CFrame}
ts:Create(part, ti, tp):Play()
                        wait(0.1)
                    end                  
                end)
            end
        else
            _G.autoPickArrow = false do
                spawn(function ()
                    while _G.autoPickArrow == false do
                        wait(1)
                  end
                end)
            end
        end
    end,
    Enabled = false
})
